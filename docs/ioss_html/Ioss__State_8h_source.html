11  /** \brief Access states for a database.
12  *
13  * All access states except STATE_INVALID, STATE_UNKNOWN, and STATE_READONLY are only
14  * for output databases.
15  */
16  enum State {
17  STATE_INVALID = -1, /**< Error state if something goes wrong. */
18  STATE_UNKNOWN, /**< Typically used at the very beginning of the database's existence
19  when the class has been created, but no reading or writing has
20  occurred. */
21  STATE_READONLY, /**< An input database that is not in STATE_UNKNOWN is in this
22  state, which means that it cannot be written to or changed. */
23  STATE_CLOSED, /**< The sates are not nested, so each state must end with a transition
24  to this state prior to entering the next state. */
25  STATE_DEFINE_MODEL, /**< Defining the metadata, which defines the topology of the model
26  (nontransient, geometry and topology). */
27  STATE_MODEL, /**< Outputting the bulk data (mesh_model_coordinates, ids, connectivity)
28  relating to the model portion. */
29  STATE_DEFINE_TRANSIENT, /**< Defining the metadata relating to the transient data. For example,
30  the element or nodal fields. */
31  STATE_TRANSIENT, /**< Outputting the transient bulk data. */
32  STATE_LAST_ENTRY
33  };
