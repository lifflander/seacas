105  DashSurfaceData(const std::vector<double> &coords, const std::vector<int> &connectivity1,
106  const std::vector<int> &connectivity2)
107  : coordinates(coords), surfaceAConnectivity(connectivity1),
108  surfaceBConnectivity(connectivity2)
109  {
110  this->setSerialDefaults();
111  }
112 
113  private:
114  void setSerialDefaults()
115  {
116  globalNumberOfNodes = coordinates.size() / SPATIAL_DIMENSION;
117 
118  globalNumberOfElementsSurface1 = surfaceBConnectivity.size() / NUM_NODES_PER_QUAD_FACE;
119  globalNumberOfElementsSurface2 = surfaceAConnectivity.size() / NUM_NODES_PER_QUAD_FACE;
120  globalNumberOfElements = globalNumberOfElementsSurface1 + globalNumberOfElementsSurface2;
121 
122  globalIdsOfLocalElements.resize(globalNumberOfElements);
123  globalIdsOfLocalNodes.resize(globalNumberOfNodes);
124 
125  for (size_t i = 0; i < globalIdsOfLocalElements.size(); i++) {
126  globalIdsOfLocalElements[i] = i + 1;
127  }
128 
129  for (size_t i = 0; i < globalIdsOfLocalNodes.size(); i++) {
130  globalIdsOfLocalNodes[i] = i + 1;
131  }
132  }
133  };
