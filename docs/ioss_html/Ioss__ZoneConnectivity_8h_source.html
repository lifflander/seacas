63  // Return number of nodes in the connection shared with the donor zone.
64  size_t get_shared_node_count() const
65  {
66  size_t snc = 1;
67  for (int i = 0; i < 3; i++) {
68  snc *= (std::abs(m_ownerRangeEnd[i] - m_ownerRangeBeg[i]) + 1);
69  }
70  return snc;
71  }
