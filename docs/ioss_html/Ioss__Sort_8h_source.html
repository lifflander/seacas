30  template <typename INT> void order3(INT v[], size_t left, size_t center, size_t right)
31  {
32  if (v[left] > v[center]) {
33  SWAP(v, left, center);
34  }
35  if (v[left] > v[right]) {
36  SWAP(v, left, right);
37  }
38  if (v[center] > v[right]) {
39  SWAP(v, center, right);
40  }
41  }

43  template <typename INT> size_t median3(INT v[], size_t left, size_t right)
44  {
45  size_t center = (left + right) / 2;
46  size_t pl = left;
47  size_t pm = center;
48  size_t pr = right;
49 
50  if (right - left > 40) {
51  size_t s = (right - left) / 8;
52  order3(v, left, left + s, left + 2 * s);
53  order3(v, center - s, center, center + s);
54  order3(v, right - 2 * s, right - s, right);
55 
56  // Now set up to get median of the 3 medians...
57  pl = left + s;
58  pm = center;
59  pr = right - s;
60  }
61  order3(v, pl, pm, pr);
62 
63  SWAP(v, center, right - 1);
64  return right - 1;
65  }

67  template <typename INT> void qsort_int(INT v[], size_t left, size_t right)
68  {
69  if (left + QSORT_CUTOFF < right) {
70  size_t pivot = median3(v, left, right);
71  size_t i = left;
72  size_t j = right - 1;
73 
74  for (;;) {
75  while (v[++i] < v[pivot]) {
76  ;
77  }
78  while (v[--j] > v[pivot]) {
79  ;
80  }
81  if (i < j) {
82  SWAP(v, i, j);
83  }
84  else {
85  break;
86  }
87  }
88 
89  SWAP(v, i, right - 1);
90  qsort_int(v, left, i - 1);
91  qsort_int(v, i + 1, right);
92  }
93  }

95  template <typename INT> void isort_int(INT v[], size_t N)
96  {
97  size_t j;
98  size_t ndx = 0;
99 
100  if (N <= 1) {
101  return;
102  }
103  INT small = v[0];
104  for (size_t i = 1; i < N; i++) {
105  if (v[i] < small) {
106  small = v[i];
107  ndx = i;
108  }
109  }
110  /* Put smallest value in slot 0 */
111  SWAP(v, 0, ndx);
112 
113  for (size_t i = 1; i < N; i++) {
114  INT tmp = v[i];
115  for (j = i; tmp < v[j - 1]; j--) {
116  v[j] = v[j - 1];
117  }
118  v[j] = tmp;
119  }
120  }
