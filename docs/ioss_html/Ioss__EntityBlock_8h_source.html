24  /** \brief Base class for all 'block'-type grouping entities, which means all
25  * members of the block are similar or have the same topology.
26  *
27  * The following derived classes are typical:
28  *
29  * -- NodeBlock -- grouping of 'similar' nodes (same degree of freedom, ...)
30  *
31  * -- ElementBlock -- grouping of 'similar' elements (same element topology,
32  * attributes, ...)
33  * 0d, 1d, 2d, 3d topology possible -- e.g., sphere, bar, quad, hex
34  */
35  class EntityBlock : public GroupingEntity
36  {
37  public:
38  Property get_implicit_property(const std::string &my_name) const override = 0;
39 
40  /** \brief Get the topology of the entities in the block.
41  *
42  * \returns The topology.
43  */
44  const ElementTopology *topology() const { return topology_; }
45 
46  /** \brief Determine whether the block contains the entity with a given id.
47  *
48  * \param[in] local_id The id to check.
49  * \returns True if the block contains the entity.
50  */
51  bool contains(size_t local_id) const
52  {
53  return idOffset < local_id && local_id <= idOffset + entityCount;
54  }
55  /** \brief Set the 'offset' for the block.
56  *
57  * The 'offset' is used to map an element location within an
58  * element block to the element 'file descriptor'.
59  * For example, the file descriptor of the 37th element in the 4th
60  * block is calculated by:
61  *
62  * file_descriptor = offset of block 4 + 37
63  *
64  * This can also be used to determine which element block
65  * an element with a file_descriptor maps into. An particular
66  * element block contains all elements in the range:
67  *
68  * offset < file_descriptor <= offset+number_elements_per_block
69  */
70  void set_offset(size_t offset) { idOffset = offset; }
71 
72  /** \brief Get the 'offset' for the block.
73  *
74  * The 'offset' is used to map an element location within an
75  * element block to the element 'file descriptor'.
76  * For example, the file descriptor of the 37th element in the 4th
77  * block is calculated by:
78  *
79  * file_descriptor = offset of block 4 + 37
80  *
81  * This can also be used to determine which element block
82  * an element with a file_descriptor maps into. An particular
83  * element block contains all elements in the range:
84  *
85  * offset < file_descriptor <= offset+number_elements_per_block
86  */
87  size_t get_offset() const { return idOffset; }
88
