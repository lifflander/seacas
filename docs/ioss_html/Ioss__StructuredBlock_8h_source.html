113  /** \brief Set the 'offset' for the block.
114  *
115  * The 'offset' is used to map a cell or node location within a
116  * structured block to the model implicit cell or node location
117  * on a single processor. zero-based.
118  *
119  * The 'global' offsets do the same except for they apply over
120  * the entire model on all processors. zero-based.
121  *
122  * For example, the file descriptor (1-based) of
123  * the 37th cell in the 4th block is calculated by:
124  *
125  * file_descriptor = offset of block 4 + 37
126  *
127  * This can also be used to determine which structured block
128  * a cell with a file_descriptor maps into. An particular
129  * structured block contains all cells in the range:
130  *
131  * offset < file_descriptor <= offset+number_cells_per_block
132  *
133  * Note that for nodes, the nodeOffset does not take into account
134  * the nodes that are shared between blocks.
135  */
136  void set_node_offset(size_t offset) { m_nodeOffset = offset; }
137  void set_cell_offset(size_t offset) { m_cellOffset = offset; }
138  void set_node_global_offset(size_t offset) { m_nodeGlobalOffset = offset; }
139  void set_cell_global_offset(size_t offset) { m_cellGlobalOffset = offset; }

210  template <typename INT_t> size_t get_cell_node_ids(INT_t *idata, bool add_offset) const
211  {
212  // Fill 'idata' with the cell node ids which are the
213  // 1-based location of each node in this zone
214  // The location is based on the "model" (all processors) zone.
215  // If this is a parallel decomposed model, then
216  // this block may be a subset of the "model" zone
217  //
218  // if 'add_offset' is true, then add the m_cellGlobalOffset
219  // which changes the location to be the location in the
220  // entire "mesh" instead of within a "zone" (all processors)
221 
222  size_t index = 0;
223  size_t offset = add_offset ? m_nodeGlobalOffset : 0;
224 
225  if (m_nk == 0 && m_nj == 0 && m_ni == 0) {
226  return index;
227  }
228 
229  for (int kk = 0; kk < m_nk + 1; kk++) {
230  size_t k = m_offsetK + kk;
231  for (int jj = 0; jj < m_nj + 1; jj++) {
232  size_t j = m_offsetJ + jj;
233  for (int ii = 0; ii < m_ni + 1; ii++) {
234  size_t i = m_offsetI + ii;
235 
236  size_t ind = k * (m_niGlobal + 1) * (m_njGlobal + 1) + j * (m_niGlobal + 1) + i;
237 
238  idata[index++] = ind + offset + 1;
239  }
240  }
241  }
242 
243  for (auto idx_id : m_globalIdMap) {
244  idata[idx_id.first] = idx_id.second;
245  }
246 
247  return index;
248  }

250  template <typename INT_t> size_t get_cell_ids(INT_t *idata, bool add_offset) const
251  {
252  // Fill 'idata' with the cell ids which are the
253  // 1-based location of each cell in this zone
254  // The location is based on the "model" zone.
255  // If this is a parallel decomposed model, then
256  // this block may be a subset of the "model" zone
257  //
258  // if 'add_offset' is true, then add the m_cellGlobalOffset
259  // which changes the location to be the location in the
260  // entire "mesh" instead of within a "zone"
261 
262  size_t index = 0;
263  size_t offset = add_offset ? m_cellGlobalOffset : 0;
264 
265  if (m_nk == 0 && m_nj == 0 && m_ni == 0) {
266  return index;
267  }
268 
269  for (int kk = 0; kk < m_nk; kk++) {
270  size_t k = m_offsetK + kk;
271  for (int jj = 0; jj < m_nj; jj++) {
272  size_t j = m_offsetJ + jj;
273  for (int ii = 0; ii < m_ni; ii++) {
274  size_t i = m_offsetI + ii;
275 
276  size_t ind = k * m_niGlobal * m_njGlobal + j * m_niGlobal + i;
277 
278  idata[index++] = ind + offset + 1;
279  }
280  }
281  }
