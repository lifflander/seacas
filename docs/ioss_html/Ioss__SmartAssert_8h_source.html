136  // allows finding if a value is of type 'const char *'
137  // and is null; if so, we cannot print it to an ostream
138  // directly!!!
139  template <class T> struct is_null_finder
140  {
141  bool is(const T & /*unused*/) const { return false; }
142  };
143 
144  template <> struct is_null_finder<char *>
145  {
146  bool is(char *const &val) { return val == nullptr; }
147  };

165  explicit Assert(const char *expr)
166  : SMART_ASSERT_A(*this), SMART_ASSERT_B(*this), needs_handling_(true)
167  {
168  context_.set_expr(expr);
169 
170  if ((logger() == nullptr) || handlers().size() < 4) {
171  // used before main!
172  Ioss::Private::init_assert();
173  }
174  }
175 
176  Assert(const Assert &other)
177  : SMART_ASSERT_A(*this), SMART_ASSERT_B(*this), context_(other.context_),
178  needs_handling_(true)
179  {
180  other.needs_handling_ = false;
181  }
182 
183  ~Assert()
184  {
185  if (needs_handling_) {
186  handle_assert();
187  }
188  }

190  template <class type> Assert &print_current_val(const type &val, const char *my_msg)
191  {
192  std::ostringstream out;
193 
194  Ioss::Private::is_null_finder<type> f;
195  bool bIsNull = f.is(val);
196  if (!bIsNull) {
197  out << val;
198  }
199  else {
200  // null string
201  out << "null";
202  }
203  context_.add_val(out.str(), my_msg);
204  return *this;
205  }

210  return *this;

255  // handles the current assertion.
256  void handle_assert()
257  {
258  logger()(context_);
259  get_handler(context_.get_level())(context_);
260  }
261 
262  /*
263  IMPORTANT NOTE:
264  The only reason logger & handlers are functions, are
265  because you might use SMART_ASSERT before main().
266 
267  In this case, since they're statics, they might not
268  be initialized. However, making them functions
269  will make it work.
270  */
271 
272  // the log
273  static assert_func &logger()
274  {
275  static assert_func inst;
276  return inst;
277  }
278 
279  // the handler
280  using handlers_collection = std::map<int, assert_func>;
281  static handlers_collection &handlers()
282  {
283  static handlers_collection inst;
284  return inst;
285  }
286 
287  static assert_func get_handler(int nLevel)
288  {
289  handlers_collection::const_iterator found = handlers().find(nLevel);
290  if (found != handlers().end()) {
291  return (*found).second;
292  }
293 
294  // we always assume the debug handler has been set
295  return (*handlers().find(lvl_debug)).second;
296  }
297 
298 private:
299  assert_context context_;
300  mutable bool needs_handling_;
