28 class ScopeGuardImplBase
29 {
30  ScopeGuardImplBase &operator=(const ScopeGuardImplBase &);
31 
32 protected:
33  ~ScopeGuardImplBase() = default;
34  ScopeGuardImplBase(const ScopeGuardImplBase &other) : dismissed_(other.dismissed_)
35  {
36  other.Dismiss();
37  }
38  template <typename J> static void SafeExecute(J &j)
39  {
40  if (!j.dismissed_) {
41  try {
42  j.Execute();
43  }
44  catch (...) {
45  }
46  }
47  }
48 
49  mutable bool dismissed_;
50 
51 public:
52  ScopeGuardImplBase() : dismissed_(false) {}
53  void Dismiss() const { dismissed_ = true; }
54 };
